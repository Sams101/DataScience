# Use Pipelines to standardise and Standardize dataset 

def create_pipeline(full_name, scaler_name, scaler, model_name, model ):
    pipelines.append((full_name,   Pipeline(  [(scaler_name, StandardScaler()),(model_name,   model)]  )))
    
scaler = StandardScaler()

pipelines = []
create_pipeline('ScaledLR', 'Scaler', scaler, 'LR', LogisticRegression() )
create_pipeline('ScaledKNN', 'Scaler', scaler, 'KNN', KNeighborsClassifier() )
create_pipeline('ScaledNB', 'Scaler', scaler, 'NB', GaussianNB() )
