Data Cleaning 

def remove_punctuation(text):
    cleaned = "".join(char for char in text if char not in ('?', '.',',', 
                                                 ';','...', ':','-','!','"',"'"))
    return cleaned

df = df.dropna(subset=['Summary'])
df = df.dropna(subset=['Summary'])
df['Summary'] = df['Summary'].apply(lambda x: x.lower() )
df['Summary'] = df['Summary'].apply(remove_punctuation)
