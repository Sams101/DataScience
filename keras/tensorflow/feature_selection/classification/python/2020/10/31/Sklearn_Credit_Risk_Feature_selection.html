# One-Hot encode features with less than 20 categorical labels. 

def one_hot_encode(df, col_name):
    new_df = pd.get_dummies(df[col_name])
    new_df = pd.concat([df, new_df], axis=1)
    new_df = new_df.drop(columns = col_name, axis=1)
    new_df = new_df.iloc[:,:-1]
    return new_df
    

data = one_hot_encode(data, 'home_ownership')

# Label Encode features with more than 20 labels. 

from sklearn.preprocessing import LabelEncoder 

def le(df, col_name):
    le = LabelEncoder()
    le.fit(df[col_name])
    df[col_name] = le.transform(df[col_name])
    return df[col_name]


data['grade'] = le(data, 'grade')
