Impute missing values 

# Create imputation function
def impute_mean(df, col_name):
    df[col_name] = df[col_name].fillna(df[col_name].mean())
    

impute_mean(df, 'mths_since_last_delinq' )
impute_mean(df, 'total_rev_hi_lim' )
impute_mean(df, 'tot_cur_bal' )
impute_mean(df, 'tot_coll_amt' )
impute_mean(df, 'collections_12_mths_ex_med' )
impute_mean(df, 'revol_util' )

# I'm replacing nan with 3000 as the user hasn't 
# had a delinquency so this is twice the length of the max delinquency.
df.mths_since_last_major_derog = df.mths_since_last_major_derog.replace(np.nan, 300)

# Convert strings to numbers and fill missing values with mean. 
df.emp_length = df.emp_length.map({'< 1 year':0.5, '1 year': 1, '2 years':2,
                                   '3 years':3,'4 years':4,'5 years':5,'6 years':6,
                                   '7 years':7,'8 years':8,'9 years':9,'10+ years':10 })
df.emp_length.fillna(df.emp_length.mean(), inplace=True)

One Hot Encode 

def one_hot(df, col_name):
    new_df = pd.get_dummies(df[col_name], prefix=col_name).iloc[:,0]
    df = pd.concat([df, new_df], axis=1)
    df.drop(columns=col_name, inplace=True)
    return df
    
df = one_hot(df, 'home_ownership')
df = one_hot(df, 'purpose')
df = one_hot(df, 'addr_state')
df = one_hot(df, 'initial_list_status')
df = one_hot(df, 'application_type')
df = one_hot(df, 'verification_status')

from sklearn.preprocessing import LabelEncoder

le = LabelEncoder()

le.fit(df.grade)
df.grade = le.transform(df.grade)

le.fit(df.sub_grade)
df.sub_grade = le.transform(df.sub_grade)

le.fit(df.term)
df.term = le.transform(df.term)

# Convert Last_week_pay

df.last_week_pay = df.last_week_pay.str.split('t', expand=True).iloc[:,0]
df.last_week_pay = df.last_week_pay.str.replace('NA',df.last_week_pay.value_counts().index[0])
df.last_week_pay = df.last_week_pay.astype(int)
